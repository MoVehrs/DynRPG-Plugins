60 void storeOriginalValues(unsigned int address, size_t length) {
61 // Skip if values already exist
62 if (hasOriginalValues(address, length)) {
63 return;
64 }
65
66 // Validate memory address range
67 if (address == 0 || address >= 0xFFFFFFFF - length) {
68 if (Debug::enableConsole) {
69 std::cout << "[DynamicQuickPatch - Memory Error]" << std::endl;
70 std::cout << "Invalid memory address: 0x" << std::hex << std::uppercase << address << std::endl;
71 std::cout << std::endl;
72 }
73 return;
74 }
75
76 try {
77 // Read current memory values
78 std::vector<unsigned char> originalBytes;
79 for (size_t i = 0; i < length; i++) {
80 originalBytes.push_back(*((unsigned char*)(address + i)));
81 }
82 originalMemoryValues[address] = originalBytes;
83
84 if (Debug::enableConsole) {
85 std::cout << "[DynamicQuickPatch - Memory]" << std::endl;
86 std::cout << "Stored original memory values at 0x" << std::hex << std::uppercase << address
87 << " (" << std::dec << length << " bytes)" << std::endl;
88 std::cout << std::endl;
89 }
90 } catch (...) {
91 if (Debug::enableConsole) {
92 std::cout << "[DynamicQuickPatch - Memory Error]" << std::endl;
93 std::cout << "Memory access violation at 0x" << std::hex << std::uppercase << address << std::endl;

107 bool restoreOriginalValues(unsigned int address) {
108 // Verify original values exist
109 if (originalMemoryValues.find(address) == originalMemoryValues.end()) {
110 return false;
111 }
112
113 // Validate memory address range
114 if (address == 0 || address >= 0xFFFFFFFF - originalMemoryValues[address].size()) {
115 if (Debug::enableConsole) {
116 std::cout << "[DynamicQuickPatch - Memory Error]" << std::endl;
117 std::cout << "Invalid memory address for restore: 0x" << std::hex << std::uppercase << address << std::endl;
118 std::cout << std::endl;
119 }
120 return false;
121 }
122
123 bool success = false;
124 try {
125 // Write original values back to memory
126 const auto& originalBytes = originalMemoryValues[address];
127 for (size_t i = 0; i < originalBytes.size(); i++) {
128 *((unsigned char*)(address + i)) = originalBytes[i];
129 }
130 success = true;
131 } catch (...) {
132 if (Debug::enableConsole) {
133 std::cout << "[DynamicQuickPatch - Memory Error]" << std::endl;
134 std::cout << "Memory access violation during restore at 0x" << std::hex << std::uppercase << address << std::endl;
135 std::cout << std::endl;
136 }
137 return false;
138 }
139
140 return success;

226 void writeHexValue(unsigned int address, const std::string& hexString) {
227 // Calculate number of bytes to write
228 size_t byteCount = hexString.length() / 2;
229
230 // Store original values if needed
231 if (!hasOriginalValues(address, byteCount)) {
232 storeOriginalValues(address, byteCount);
233 }
234
235 // Validate memory address range
236 if (address == 0 || address >= 0xFFFFFFFF - byteCount) {
237 if (Debug::enableConsole) {
238 std::cout << "[DynamicQuickPatch - Memory Error]" << std::endl;
239 std::cout << "Invalid memory address for hex write: 0x" << std::hex << std::uppercase << address << std::endl;
240 std::cout << std::endl;
241 }
242 return;
243 }
244
245 try {
246 // Convert hex string to bytes and write to memory
247 for (size_t i = 0; i < hexString.length(); i += 2) {
248 // Skip incomplete byte pairs
249 if (i + 1 >= hexString.length()) break;
250
251 // Convert hex pair to byte value
252 std::string byteStr = hexString.substr(i, 2);
253 unsigned char byte = (unsigned char)strtol(byteStr.c_str(), NULL, 16);
254
255 // Write byte to memory
256 *((unsigned char*)(address + (i/2))) = byte;
257 }
258 } catch (...) {
259 if (Debug::enableConsole) {
260 std::cout << "[DynamicQuickPatch - Memory Error]" << std::endl;
261 std::cout << "Memory access violation during hex write at 0x" << std::hex << std::uppercase << address << std::endl;
262 std::cout << std::endl;

291 std::cout << "Variable: " << variableId << std::endl;
292 std::cout << "Address: 0x" << std::hex << std::uppercase << address << std::endl;
293 std::cout << "Valid range: " << std::dec << DQP_INT8_MIN << " to " << DQP_INT8_MAX << std::endl;
294 std::cout << "Value will be clamped to fit in range." << std::endl;
295 std::cout << std::endl;
296 }
297 }
298 break;
299
300 case DynamicQuickPatchConfig::QPTYPE_32BIT:
301 // Check 32-bit signed range
302 if (value < DQP_INT32_MIN || value > DQP_INT32_MAX) {
303 isValid = false;
304 if (Debug::enableConsole) {
305 std::cout << "[DynamicQuickPatch - Range Error]" << std::endl;
306 std::cout << "Value " << value << " is out of range for 32-bit type!" << std::endl;
307 std::cout << "Variable: " << variableId << std::endl;
308 std::cout << "Address: 0x" << std::hex << std::uppercase << address << std::endl;
309 std::cout << "Value may cause undefined behavior." << std::endl;
310 std::cout << std::endl;
311 }
312 }
313 break;
314
315 case DynamicQuickPatchConfig::QPTYPE_HEX_RAW:
316 // No range validation for hex values
317 break;
318 }

334 void updateQuickPatch(const DynamicQuickPatchConfig::QuickPatchMapping& mapping, int value) {
335 // Handle patch deactivation - only for hex patches
336 if (value == 0 && mapping.type == DynamicQuickPatchConfig::QPTYPE_HEX_RAW) {
337 // Determine patch size for hex type
338 size_t size = mapping.hexValue.length() / 2;
339 std::string typeStr = "Raw Hex";
340
341 // Restore original memory values if available
342 bool restored = false;
343 if (hasOriginalValues(mapping.address, size)) {
344 restored = restoreOriginalValues(mapping.address);
345 }
346
347 if (Debug::enableConsole) {
348 std::cout << "[DynamicQuickPatch - Patch Disabled]" << std::endl;
349 std::cout << "QuickPatch Disabled" << std::endl;
350 std::cout << "Variable: " << mapping.variableId << std::endl;
351 std::cout << "Address: 0x" << std::hex << std::uppercase << mapping.address << std::endl;
352 std::cout << "Type: " << typeStr << std::endl;
353 if (restored) {
354 std::cout << "Original memory values restored (variable = 0)" << std::endl;
355 } else {
356 std::cout << "Patch is now inactive (variable = 0)" << std::endl;
357 std::cout << "Note: Original values were not available to restore" << std::endl;
358 }
359 std::cout << std::endl;
360 }
361 return;
362 }
363
364 // Read current memory value for logging
365 std::string oldValueStr;
366 switch (mapping.type) {
367 case DynamicQuickPatchConfig::QPTYPE_8BIT: {
368 char oldVal = *((char*)mapping.address);
369 oldValueStr = "%" + std::to_string((int)oldVal);
370 break;
371 }
372 case DynamicQuickPatchConfig::QPTYPE_32BIT: {
373 int oldVal = *((int*)mapping.address);
374 oldValueStr = "#" + std::to_string(oldVal);
375 break;
376 }
377 case DynamicQuickPatchConfig::QPTYPE_HEX_RAW: {
378 // Format current memory as hex string
379 std::string hex;
380 size_t hexLen = mapping.hexValue.length();
381 for (size_t i = 0; i < hexLen; i += 2) {
382 unsigned char byte = *((unsigned char*)(mapping.address + (i / 2)));
383 char buf[3];
384 sprintf(buf, "%02X", byte);
385 hex += buf;
386 }
387 oldValueStr = hex;
388 break;
389 }
390 }
391
392 // Validate and adjust value if needed (for non-hex types)
393 if (mapping.type != DynamicQuickPatchConfig::QPTYPE_HEX_RAW) {
394 validateValue(value, mapping.type, mapping.variableId, mapping.address);
395 }
396 int adjustedValue = value;
397
398 // Apply new value with range clamping
399 switch (mapping.type) {
400 case DynamicQuickPatchConfig::QPTYPE_8BIT:
401 // Clamp value to 8-bit range (0 is valid)
402 if (adjustedValue < DQP_INT8_MIN) adjustedValue = DQP_INT8_MIN;
403 if (adjustedValue > DQP_INT8_MAX) adjustedValue = DQP_INT8_MAX;
404 write8bitValue(mapping.address, adjustedValue);
405 break;
406
407 case DynamicQuickPatchConfig::QPTYPE_32BIT:
408 // Direct write (0 is valid)
409 write32bitValue(mapping.address, adjustedValue);
410 break;
411
412 case DynamicQuickPatchConfig::QPTYPE_HEX_RAW:
413 // For hex patches, write the predefined hex value
414 writeHexValue(mapping.address, mapping.hexValue);
415 break;
416 }
417
418 // Log memory update details
419 if (Debug::enableConsole) {
420 // Format display values
421 std::string typeStr;
422 std::string newValueStr;
423 std::string rangeInfo;
424
425 switch (mapping.type) {
426 case DynamicQuickPatchConfig::QPTYPE_8BIT:
427 typeStr = "8-bit (%)";
428 newValueStr = "%" + std::to_string(adjustedValue);
429 if (value != adjustedValue) {
430 rangeInfo = "\nOriginal value: " + std::to_string(value) + " (clamped to range)";
431 }
432 break;
433
434 case DynamicQuickPatchConfig::QPTYPE_32BIT:
435 typeStr = "32-bit (#)";
436 newValueStr = "#" + std::to_string(adjustedValue);
437 break;
438
439 case DynamicQuickPatchConfig::QPTYPE_HEX_RAW:
440 typeStr = "Raw Hex";
441 newValueStr = mapping.hexValue;
442 rangeInfo = "\nPatch is now active (variable > 0)";
443 break;
444 }
445
446 std::cout << "[DynamicQuickPatch - Memory Update]" << std::endl;
447 std::cout << "QuickPatch Updated" << std::endl;
448 std::cout << "Variable: " << mapping.variableId << std::endl;
449 std::cout << "Address: 0x" << std::hex << std::uppercase << mapping.address << std::endl;
450 std::cout << "Type: " << typeStr << std::endl;
451 std::cout << "Old Value: " << oldValueStr << std::endl;
452 std::cout << "New Value: " << newValueStr << std::endl;
453 if (!rangeInfo.empty()) {
454 std::cout << rangeInfo << std::endl;
455 }
456 std::cout << std::endl;
457 }

491 void onNewGame() {
492 // Track restoration results
493 int restoredCount = 0;
494 int failedCount = 0;
495
496 // Get unique addresses from mappings
497 const auto& mappings = DynamicQuickPatchConfig::getMappings();
498 std::set<unsigned int> processedAddresses;
499
500 for (const auto& mapping : mappings) {
501 // Skip already processed addresses
502 if (processedAddresses.find(mapping.address) != processedAddresses.end()) {
503 continue;
504 }
505
506 // Mark address as processed
507 processedAddresses.insert(mapping.address);
508
509 // Calculate patch size based on type
510 size_t size = 0;
511 switch (mapping.type) {
512 case DynamicQuickPatchConfig::QPTYPE_8BIT:
513 size = 1;
514 break;
515
516 case DynamicQuickPatchConfig::QPTYPE_32BIT:
517 size = 4;
518 break;
519
520 case DynamicQuickPatchConfig::QPTYPE_HEX_RAW:
521 size = mapping.hexValue.length() / 2;
522 break;
523 }
524
525 // Attempt to restore original values
526 if (hasOriginalValues(mapping.address, size)) {
527 bool success = restoreOriginalValues(mapping.address);
528 if (success) {
529 restoredCount++;
530 } else {
531 failedCount++;
532 }
533 }
534 }
535
536 // Log restoration results
537 if (Debug::enableConsole && (restoredCount > 0 || failedCount > 0)) {
538 std::cout << "[DynamicQuickPatch - Memory Reset]" << std::endl;
539 std::cout << "Restored " << restoredCount << " memory locations" << std::endl;
540 if (failedCount > 0) {
541 std::cout << "Failed to restore " << failedCount << " memory locations" << std::endl;
542 }
543 std::cout << std::endl;
544 }
545
546 // Clear stored memory values
547 originalMemoryValues.clear();
548 }

588
589 // Update patches configured for load game
590 const auto& mappings = DynamicQuickPatchConfig::getMappings();
591 int appliedCount = 0;
592 int skippedCount = 0;
593
594 for (const auto& mapping : mappings) {
595 if (mapping.applyOnLoadGame) {
596 // Validate variable ID range
597 if (mapping.variableId > 0 && mapping.variableId <= DynamicQuickPatchConfig::getMaxVariableId()) {
598 int value = RPG::variables[mapping.variableId];
599 updateQuickPatch(mapping, value);
600 appliedCount++;
601 } else {
602 if (Debug::enableConsole) {
603 std::cout << "[DynamicQuickPatch - Load Game Error]" << std::endl;
604 std::cout << "Invalid variable ID: " << mapping.variableId << std::endl;
605 std::cout << std::endl;
606 }
607 }
608 } else {
609 skippedCount++;
610 }
611 }
612
613 // Log patch update summary
614 if (Debug::enableConsole && (appliedCount > 0 || skippedCount > 0)) {
615 std::cout << "[DynamicQuickPatch - Load Game Summary]" << std::endl;
616 std::cout << "Applied " << appliedCount << " patches on load game." << std::endl;
617 std::cout << "Skipped " << skippedCount << " patches (OnLoadGame=false)." << std::endl;
618 std::cout << std::endl;

620
